259static void *dynarray_growf(void *a, long cap, long membsize) {
260 if (a == NULL) {
261 a = ((struct _dynarray_info *)DYNARRAY_MALLOC(
262 sizeof(struct _dynarray_info) +
263 membsize * DYNARRAY_MIN_CAPACITY)) +
264 1;
265 dynarray_info(a)->capacity = DYNARRAY_MIN_CAPACITY;
266 dynarray_info(a)->length = 0;
267 }
268 if (cap <= dynarray_cap(a))
269 return a;
270 void *b;
271 b = ((struct _dynarray_info *)DYNARRAY_REALLOC(
272 dynarray_info(a), sizeof(struct _dynarray_info) + cap * membsize)) +
273 1;
274 dynarray_info(b)->capacity = cap;
275 return b;
276}
