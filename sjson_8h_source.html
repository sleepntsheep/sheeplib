282 static sjson_resultnum sjsonlexer_lexnumber(sjsonlexer *lexer) {
283  const char *numberstart = lexer->c;
284  bool didpoint = false, didsign = false;
285  sjsonbuf buf;
286  sjsonbuf_init(&buf);
287 
288  while (!sjsonlexer_isend(lexer)) {
289  char c = sjsonlexer_peek(lexer);
290  if (!isdigit(c)) {
291  if (c == '+' || c == '-') {
292  if (didsign) {
293  goto bad;
294  }
295  didsign = true;
296  }
297  else if (c == '.') {
298  if (didpoint) {
299  goto bad;
300  }
301  didpoint = true;
302  }
303  else {
304  break;
305  }
306  }
307  sjsonlexer_advance(lexer);
308  }
309  lexer->c--;
310 
311  /* avoid modification of original source */
312  sjsonbuf_push(&buf, numberstart, lexer->c - numberstart + 1);
313 
314  sjsonlexer_pushtok(lexer, SJSON_TKNUMBERLITERAL, buf.buf, lexer->c);
315 
316  return SJSON_SUCCESS;
317 bad:
318  return SJSON_ERR_INVALID_SOURCE;
319 }

321 static sjson_resultnum sjsonlexer_lexstring(sjsonlexer *lexer) {
322  if (sjsonlexer_advance(lexer) != '\"') {
323  return SJSON_ERR_WRONG_TYPE;
324  }
325  const char *stringend = lexer->c;
326 
327  /* find terminating double quote, need to escape \" */
328  while (stringend[0] != '\"') {
329  if (stringend > lexer->end) {
330  return SJSON_ERR_NO_TERMINATING_DOUBLE_QUOTE;
331  }
332  if (stringend[0] == '\\')
333  stringend++;
334  stringend++;
335  }
336 
337  /* can't just push new token, escape character reasons */
338  sjsonbuf buf;
339  sjsonbuf_init(&buf);
340  while (sjsonlexer_peek(lexer) != '"'
341  && lexer->c < stringend) {
342  char ch = sjsonlexer_advance(lexer);
343  if (ch == '\\') {
344  int advance_ret;
345  if ((advance_ret = sjsonlexer_advance(lexer)) == -1) {
346  return SJSON_ERR_NO_TERMINATING_DOUBLE_QUOTE;
347  }
348  /* deal with escape character */
349  switch (advance_ret) {
350  case '\\':
351  sjsonbuf_push(&buf, "\\", 1);
352  break;
353  case 'n':
354  sjsonbuf_push(&buf, "\n", 1);
355  break;
356  case 'f':
357  sjsonbuf_push(&buf, "\f", 1);
358  break;
359  case 'r':
360  sjsonbuf_push(&buf, "\r", 1);
361  break;
362  case 't':
363  sjsonbuf_push(&buf, "\t", 1);
364  break;
365  case '\"':
366  sjsonbuf_push(&buf, "\"", 1);
367  break;
368  case '/':
369  sjsonbuf_push(&buf, "/", 1);
370  break;
371  case 'u':
372  /* WILL IMPLEMENT LATER, HATETHIS */
373  break;
374  }
375  } else {
376  sjsonbuf_push(&buf, &ch, 1);
377  }
378  }
379  sjsonlexer_pushtok(lexer, SJSON_TKSTRINGLITERAL, buf.buf, buf.buf + buf.len);
380  return SJSON_SUCCESS;
381 }

412 sjson_resultnum sjsonlexer_lex(sjsonlexer *lexer) {
413  while (!sjsonlexer_isend(lexer)) {
414  int ret;
415  switch (sjsonlexer_peek(lexer)) {
416  case '[':
417  sjsonlexer_pushtok(lexer, SJSON_TKLSQUAREBRACKET, lexer->c, lexer->c);
418  break;
419  case ']':
420  sjsonlexer_pushtok(lexer, SJSON_TKRSQUAREBRACKET, lexer->c, lexer->c);
421  break;
422  case '{':
423  sjsonlexer_pushtok(lexer, SJSON_TKLBRACE, lexer->c, lexer->c);
424  break;
425  case '}':
426  sjsonlexer_pushtok(lexer, SJSON_TKRBRACE, lexer->c, lexer->c);
427  break;
428  case ',':
429  sjsonlexer_pushtok(lexer, SJSON_TKCOMMA, lexer->c, lexer->c);
430  break;
431  case ':':
432  sjsonlexer_pushtok(lexer, SJSON_TKCOLON, lexer->c, lexer->c);
433  break;
434  case '0':
435  case '1':
436  case '2':
437  case '3':
438  case '4':
439  case '5':
440  case '6':
441  case '7':
442  case '8':
443  case '9':
444  case '+':
445  case '-':
446  if ((ret = sjsonlexer_lexnumber(lexer))) {
447  return ret;
448  }
449  break;
450  case '\"':
451  if ((ret = sjsonlexer_lexstring(lexer))) {
452  return ret;
453  }
454  break;
455  case 'n':
456  case 't':
457  case 'f':
458  if ((ret = sjsonlexer_lexprimitive(lexer))) {
459  return ret;
460  }
461  break;
462  case ' ':
463  case '\t':
464  case '\n':
465  break;
466  default:
467  logger("unknown char");
468  break;
469  }
470  sjsonlexer_advance(lexer);
471  }
472  return SJSON_SUCCESS;
473 }

496 static sjson_result sjson_parseobject(sjsontokarr *toks) {
497  sjson_result obj = sjson_new(SJSON_OBJECT);
498  if (obj.err) {
499  return (sjson_result) {
500  .err = obj.err
501  };
502  }
503 
504  if (sjsontokarr_advance(toks).type != SJSON_TKLBRACE) {
505  return (sjson_result) {
506  .err = SJSON_ERR_INVALID_SOURCE
507  };
508  }
509 
510 
511  while (sjsontokarr_peek(toks).type != SJSON_TKRBRACE) {
512  const char *key = sjsontokarr_advance(toks).start;
513  sjson_result child;
514  if (sjsontokarr_advance(toks).type != SJSON_TKCOLON) {
515  return (sjson_result) {
516  .err = SJSON_ERR_NO_TERMINATING_BRACE
517  };
518  }
519  child = sjson_parse(toks);
520  if (child.err) {
521  return child;
522  }
523  child.json->key = key;
524  int ret = sjson_addchild(obj.json, child.json);
525  if (ret != SJSON_SUCCESS) {
526  return (sjson_result) {
527  .err = ret
528  };
529  }
530  sjsontok next = sjsontokarr_peek(toks);
531  if (next.type == SJSON_TKCOMMA) {
532  sjsontokarr_advance(toks);
533  }
534  else if (next.type != SJSON_TKRBRACE) {
535  return (sjson_result) {
536  .err = SJSON_ERR_NO_TERMINATING_BRACE,
537  };
538  }
539  }
540  return obj;
541 }

576 sjson_result sjson_parse(sjsontokarr *toks) {
577  sjson_result ret;
578  switch (sjsontokarr_peek(toks).type) {
579  case SJSON_TKTRUE:
580  ret = sjson_new(SJSON_TRUE);
581  break;
582  case SJSON_TKFALSE:
583  ret = sjson_new(SJSON_FALSE);
584  break;
585  case SJSON_TKNULL:
586  ret = sjson_new(SJSON_NULL);
587  break;
588  case SJSON_TKLBRACE:
589  ret = sjson_parseobject(toks);
590  if (ret.err) {
591  return ret;
592  }
593  break;
594  case SJSON_TKLSQUAREBRACKET:
595  ret = sjson_parsearray(toks);
596  if (ret.err) {
597  return ret;
598  }
599  break;
600  case SJSON_TKNUMBERLITERAL:
601  ret = sjson_new(SJSON_NUMBER);
602  if (ret.err) {
603  return ret;
604  }
605  ret.json->v.num = strtod(sjsontokarr_peek(toks).start, NULL);
606  break;
607  case SJSON_TKSTRINGLITERAL:
608  ret = sjson_new(SJSON_STRING);
609  if (ret.err) {
610  return ret;
611  }
612  ret.json->v.str = sjsontokarr_peek(toks).start;
613  break;
614  default:
615  logger("parse: unknown token %s %s", sjson_tokennames, sjsontokarr_peek(toks).type, sjsontokarr_peek(toks).start);
616  return (sjson_result) {
617  .err = SJSON_ERR_UNKNOWN_TOKEN,
618  };
619  }
620  sjsontokarr_advance(toks);
621  return ret;
622 }

624 sjson_resultnum sjson_deletechild(sjson *json, sjson *child) {
625  if (json->v.child == NULL) {
626  return false;
627  }
628  if (json->v.tail == child) {
629  if (json->v.child == child) {
630  json->v.tail = json->v.child = NULL;
631  }
632  else {
633  json->v.tail = child->prev;
634  }
635  } else if (json->v.child == child) {
636  json->v.child = child->next;
637  }
638  if (child->prev) {
639  child->prev->next = child->next;
640  }
641  if (child->next) {
642  child->next->prev = child->prev;
643  }
644  return SJSON_SUCCESS;
645 }
