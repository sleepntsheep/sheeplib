5/* dynarray.h - v0.02 - sleepntsheep 2022
6 * dynarray.h is single-header library for
7 * nice dynamic array in C, similar to
8 * std::vector in C++
9 *
10 * inspired by https://github.com/nothings/stb/blob/master/stb_ds.h
11 *
12 * Instruction & How to use
13 *
14 * #define SHEEP_DYNARRAY_IMPLEMENTATION
15 * #include "dynarray.h"
16 * T* arr = dynarray_new(T);
17 * dynarray_push(arr, X);
18 *
19 * If you don't want arrfnc shorthand
20 * to dynarray_fnc then define SHEEP_DYNARRAY_NOSHORTHAND
21 * to avoid name collision
22 *
23 * If you want to use custom alloactor,
24 * define DYNARRAY_MALLOC and DYNARRAY_REALLOC to them
25 * before including dynarray.h
26 *
27 * How it works
28 *
29 * Similar to stb_ds.h, this allocates extra memory for
30 * type struct _dynarray_info in front of the array which
31 * act as a header that store information about the array,
32 * specifically length, capacity, membsize.
33 *
34 *
35 * As opposed too using NULL to create new dynarray
36 * and checking for NULL reference on every function/macro
37 * like in stb_ds.h
38 * I decided to have dynarray_new macro for creating dynarray
39 * While the redundance type may look a bit ugly
40 * It make the code clearer that arr is dynamic array,
41 * not just normal pointer
42 * And it make dynarray.h more maintainable
43 *
44 * Notes :
45 * Due to limitation of the C language,
46 * almost all macros here are not type-safe
47 * this is a hacky solution for dynamic array,
48 * but it provide nicer interface than other approach
49 * for example arr[i] rather than vec_at(arr, i)
50 * moreover it can be used for any type
51 *
52 * Documentations
53 *
54 * arrnew(T) :
55 * create and return new dynamic array with type T
56 *
57 * arrfree(T) :
58 * free dynamic array
59 *
60 * arrpush(A, x) :
61 * grow array if needed and
62 * push x into the end of array A
63 *
64 * arrpop(A) :
65 * remove last element of array A and return it
66 *
67 * arrback(A) :
68 * return last element of array, without removing it
69 *
70 * arrlen(A) :
71 * return length of array A
72 *
73 * arrcap(A) :
74 * return capacity of array A
75 *
76 * arrins(A, idx, x) :
77 * grow array if needed and
78 * insert x into index idx of array A
79 * moving all following elements one index to the right
80 *
81 * arrdel(A, idx) :
82 * remove element at index idx of array A
83 * moving all following elements one index to the left
84 *
85 * arrsetlen(A, len) :
86 * set length of array A to *at least* len
87 * allocating more memory if needed
88 * *memory will be uninitialized
89 *
90 * arrsetcap(A, cap) :
91 * set capacity of array A to *at least* cap
92 * allocating more memory if needed
93 * *memory will be uninitialized
94 *
95 * arrsort *won't* be implemented, just use qsort from stdlib
96 * this library *cannot* be used in C++,
97 * due to C++ not allowing implicit pointer conversion
98 * you shouldn't use this in C++ anyway, use std::vector
99 * */

222static void *dynarray_growf(void *a, long cap, long membsize) {
223 if (!a) {
224 a = ((struct _dynarray_info*)
225 DYNARRAY_MALLOC(sizeof(struct _dynarray_info)
226 + membsize * DYNARRAY_MIN_CAPACITY))+1;
227 dynarray_info(a)->capacity = DYNARRAY_MIN_CAPACITY;
228 dynarray_info(a)->length = 0;
229 }
230 if (cap <= dynarray_cap(a))
231 return a;
232 void *b;
233 b = ((struct _dynarray_info*)
234 DYNARRAY_REALLOC(dynarray_info(a), sizeof(struct _dynarray_info) + cap * membsize))+1;
235 dynarray_info(b)->capacity = cap;
236 return b;
237}
